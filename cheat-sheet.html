Ruby Fundamentals

Arrays

                      array = []
                      array = [1, 2, 3, 4]

Adding a Value to an Array

                      array = [1, 2, 3, 4]
                      array.push(5)
                      #=> array = [1, 2, 3, 4, 5]

Reversing an Array

                      array = [1, 2, 3, 4]
                      array.reverse
                      #=> array = [5, 4, 3, 2, 1]

Joining Values in Array

                      array = [1, 2, 3, 4]
                      array.join
                      #=> array = 12345

Determining Length of Array

                      array = [1, 2, 3, 4]
                      array.length
                      #=> 4

Iterating Through an Array

                      array = [1, 2, 3, 4]
                      array.map { |x| x + 1}
                      #=> array = [2, 3, 4, 5]

JavaScript Fundamentals

Arrays

                        var array = [];
                        var array = [1, 2, 3, 4];

Adding a Value to an Array

                        var array = [1, 2, 3, 4];
                        array.push("5");
                        // → array = [1, 2, 3, 4, 5];

Reversing an Array

                        var array = [1, 2, 3, 4];
                        array.reverse();
                        // → array = [5, 4, 3, 2, 1];

Joining Values in Array

                        var array = [1, 2, 3, 4];
                        var number = array.join(" ");
                        // → number = 12345;

Determining Length of Array

                        var array = [1, 2, 3, 4];
                        array.length();
                        // → 4;

Iterating Through an Array

                        var array = [1, 2, 3, 4];
                        var addition = array.map(function(num){
                          return num + 1;
                         });
                        // → addition = [2, 3, 4, 5];
